238     // Note: this is a little dangerous as it makes an assumption that we want to cache every index for every pps
239     private void cachePpsIndexIfSupported(String key) {
240         Object o = SessionUtil.getSessionProperty(key);
241         if (o instanceof PagerPaginationSupport) {
242             PagerPaginationSupport pps = (PagerPaginationSupport)o;
243             SessionUtil.setSessionProperty(key + INDEX_SUFFIX, new Integer(pps.getStartIndex()));
244         }
245     }

247     // called by config
248     public static void clearCacheIncludingIndexes() {
249         staticlog.debug("Clearing all cache (removing all session data with keys that start with '" + PLUGIN_SESSION_KEY_PREFIX + DELIMITER + "')");
250         Map session = (Map) ActionContext.getContext().get("session");
251         Iterator iter = session.keySet().iterator();
252         while (iter.hasNext()) {
253             String key = (String)iter.next();
254             // intentionally not removing index here
255             if (key.startsWith(PLUGIN_SESSION_KEY_PREFIX + DELIMITER)) {
256                 session.remove(key);
257             }
258         }
259     }

261     // Note: intentionally not clearing index cache. That is used to find out where you were in result set.
262     public void clearCache() {
263         log.debug("Clearing all cache (removing all session data with keys that start with '" + PLUGIN_SESSION_KEY_PREFIX + "' and not ending with '" + INDEX_SUFFIX + "')");
264         Map session = (Map) ActionContext.getContext().get("session");
265         Iterator iter = session.keySet().iterator();
266         while (iter.hasNext()) {
267             String key = (String)iter.next();
268             // intentionally not removing index here
269             if (key.startsWith(PLUGIN_SESSION_KEY_PREFIX + DELIMITER) && !key.endsWith(INDEX_SUFFIX)) {
270                 cachePpsIndexIfSupported(key);
271                 session.remove(key);
272             }
273         }
274     }

308     public String getCacheAsHtml() {
309         StringBuffer sb = new StringBuffer();
310         sb.append("<table><tr><td>Key</td><td>Data</td></tr>");
311         try {
312             TreeMap session = new TreeMap(SessionUtil.getSession());
313             Iterator iter = session.keySet().iterator();
314             boolean flag = false;
315             while (iter.hasNext()) {
316 
317                 String key = (String)iter.next();
318 
319                 // intentionally not removing index here
320                 if (key.startsWith(PLUGIN_SESSION_KEY_PREFIX + DELIMITER)) {
321                     if ( flag ) {
322                         sb.append("<tr>");
323                     }
324                     else {
325                         sb.append("<tr class=\"even\">");
326                     }
327                     sb.append("<td>" + key + "</td>");
328                     Object val = session.get(key);
329                     if (val instanceof PagerPaginationSupport) {
330                         sb.append("<td>total size: " + ((PagerPaginationSupport)val).getTotal() + "</td>");
331                     }
332                     else if (val instanceof Integer) {
333                         sb.append("<td>" + ((Integer)val).intValue() + "</td>");
334                     }
335                     else {
336                         sb.append("<td>" + val + "</td>");
337                     }
338                     sb.append("</tr>");
339 
340                     flag = !flag;
341                 }
342 
343             }
344         }
345         catch (Throwable t) {
346             LogUtil.errorWithRemoteUserInfo(log, "Problem debugging action", t);
347         }
348         sb.append("</tr></table>");
349         return sb.toString();
350     }
