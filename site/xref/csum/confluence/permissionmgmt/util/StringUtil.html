44  public class StringUtil {
45  
46      private static final Log log = LogFactory.getLog(StringUtil.class);

48      public static final String CHARACTERS_TO_CLEAN = "[<>/]";
49      public static final String DELIMITING_CHARACTERS = "[:;,]";
50  
51      public static String clean( String s ) {
52          return s.replaceAll(CHARACTERS_TO_CLEAN, "");
53      }
54  
55      public static boolean isNullOrEmpty(String s) {
56          if (s == null || "".equals(s)) {
57              return true;
58          }
59          return false;
60      }
61  
62      public static List getCleanedListFromDelimitedValueString( String s ) {
63          List result = null;
64          if ( s != null ) {
65              result = new ArrayList();
66              s = s.replaceAll(CHARACTERS_TO_CLEAN, "");
67              String[] valueArray = s.split(DELIMITING_CHARACTERS);
68              for (int i=0; i<valueArray.length; i++) {
69                  String value = valueArray[i];
70                  if (value!=null) {
71                      value = value.trim();
72                      if (!"".equals(value)) {
73                          result.add(value);
74                      }
75                  }
76              }
77          }
78          return result;
79      }
80  
81      public static String convertCollectionToCommaDelimitedString( Collection collection ) {
82          StringBuffer sb = new StringBuffer();
83          if ( collection != null ) {
84              Iterator iter = collection.iterator();
85              int count = 0;
86              while (iter.hasNext()) {
87                  String item = (String)iter.next();
88                  if (count!=0) {
89                      sb.append(", ");
90                  }
91                  sb.append(item);
92                  count++;
93              }
94          }
95          return sb.toString();
96      }
97  
98      public static boolean areBothNullOrAreEqual(String s1, String s2) {
99          boolean result = false;
100         if (s1==s2) {
101             // note this checks for null==null as well as instance ref is same as instance ref
102             result = true;
103         }
104         else if (s1==null || s2==null) {
105             //implied result = false;
106         }
107         else if (s1.equals(s2)) {
108             return true;
109         }
110         return result;
111     }
