48  /***
49   * @author Gary S. Weaver
50   */
51  public class LDAPHelper
52  {
53      private static final Log log = LogFactory.getLog(LDAPHelper.class);
54  
55  	public static LDAPUser getLDAPUser(CustomPermissionConfigurable config, String userid) throws ParserConfigurationException,
56  			LDAPException
57  	{
58          Properties builtConfig=buildConfig(config);
59          LDAPUser u = null;
60  
61          // this lock is on a static object (this class, not the instance) so that it is JVM-wide
62          // error this is intended to fix: "SAX problem: FWK005 parse may not be called while parsing."
63          // error case: http://developer.atlassian.com/jira/browse/SUSR-58
64          // TODO: ldaputil or the libraries it uses need to be fixed to avoid locking like this
65          synchronized (LDAPHelper.class) {
66              LDAPLookupUtil lookup = new LDAPLookupUtil(builtConfig);
67              u = lookup.getUserDetails(userid);
68          }
69          return u;
70  	}
71  	
72  	private static Properties buildConfig(CustomPermissionConfigurable config)
73  	{
74  		Properties p=new Properties();
75  
76  		String narrowingFilterExpression = config.getLdapNarrowingFilterExpression();
77          if ( narrowingFilterExpression != null && !"".equals(narrowingFilterExpression.trim())) {
78              // example "(objectclass=user)"
79              p.setProperty(LDAPLookupUtil.LDAPUTIL_LDAP_NARROWING_FILTER_EXPRESSION, narrowingFilterExpression);
80          }                
81  
82  		String userFullNameFormat = config.getUserFullNameFormat();
83  		boolean lastCommaFirstFormat = (userFullNameFormat != null) && (userFullNameFormat.equals(CustomPermissionConfigConstants.USER_FULL_NAME_FORMAT_TYPE_LASTNAME_COMMA_FIRSTNAME));
84  		boolean firstLastFormat = (userFullNameFormat != null) && (userFullNameFormat.equals(CustomPermissionConfigConstants.USER_FULL_NAME_FORMAT_TYPE_FIRSTNAME_LASTNAME));
85  		boolean idFormat = (userFullNameFormat != null) && (userFullNameFormat.equals(CustomPermissionConfigConstants.USER_FULL_NAME_FORMAT_TYPE_ID));
86  
87  		if (lastCommaFirstFormat)
88  		{
89  			p.setProperty(LDAPLookupUtil.LDAPUTIL_LDAP_NAME_FORMAT, "0");
90  		}
91  		else if (firstLastFormat)
92  		{
93  			p.setProperty(LDAPLookupUtil.LDAPUTIL_LDAP_NAME_FORMAT, "1");
94  		}
95  		else if (idFormat)
96  		{
97  			p.setProperty(LDAPLookupUtil.LDAPUTIL_LDAP_NAME_FORMAT, "2");
98  		}
99  		else
100 		{
101 			LogUtil.warnWithRemoteUserInfo(log, "Unknown user fullname format '" + userFullNameFormat + "'. Setting fullname format to 2");
102 			p.setProperty(LDAPLookupUtil.LDAPUTIL_LDAP_NAME_FORMAT, "2");
103 		}
104 
105 		// make all userids lowercase (the Confluence API seems to enforce lowercase for usernames and groupnames)
106         p.setProperty(LDAPLookupUtil.LDAPUTIL_LDAP_FORCE_USERID_CASE, "0");
107         // no need yet to make this configurable
108         p.setProperty(LDAPLookupUtil.LDAPUTIL_LDAP_CONTROL_SUBTREE_SCOPE, "2");
109 
110 		// User selection would choose either LDAPLookup.OSUSER_PROVIDER or LDAPLookup.ATLASSIAN_USER_PROVIDER
111 		String providerType = config.getProviderType();
112 		if (CustomPermissionConfigConstants.PROVIDER_TYPE_ATLASSIAN_USER.equals(providerType))
113 		{
114 			log.debug("Using atlassian-user as provider type for LDAP config. Provider type from config was " + providerType);
115 			p.setProperty(LDAPLookupUtil.LDAPUTIL_PROVIDER_TYPE, LDAPLookupUtil.LDAPUTIL_PROVIDER_ATLASSIAN_USER);
116 		}
117 		else
118 		{
119 			log.debug("Using osuser as provider type for LDAP config since it didn't match " +
120 					CustomPermissionConfigConstants.PROVIDER_TYPE_ATLASSIAN_USER + ". Provider type from config was " +
121 					providerType);
122 			p.setProperty(LDAPLookupUtil.LDAPUTIL_PROVIDER_TYPE, LDAPLookupUtil.LDAPUTIL_PROVIDER_OSUSER);
123 			p.setProperty(LDAPLookupUtil.LDAPUTIL_OSUSER_PROVIDERCLASS, config.getLdapProviderFullyQualifiedClassname());
124 			p.setProperty(LDAPLookupUtil.LDAPUTIL_LDAP_ID_ATTRIBUTE, config.getLdapUserIdAttribute());
125 			p.setProperty(LDAPLookupUtil.LDAPUTIL_LDAP_EMAIL_ATTRIBUTE, config.getLdapEmailAttribute());
126 			p.setProperty(LDAPLookupUtil.LDAPUTIL_LDAP_FIRST_NAME_ATTRIBUTE, config.getLdapFirstNameAttribute());
127 			p.setProperty(LDAPLookupUtil.LDAPUTIL_LDAP_LAST_NAME_ATTRIBUTE, config.getLdapLastNameAttribute());
128 		}
129 		
130 		if (log.isInfoEnabled())
131 		{
132 			Enumeration e = p.keys();
133 			do
134 			{
135 				String key=(String)e.nextElement();
136 				log.info("LDAP Env: "+key+" = "+p.getProperty(key));
137 			} while (e.hasMoreElements());
138 		}
139 
140 		return p;
141 	}
