48  public class LazyLoadingUserByUsernamePager implements Pager {
49  
50      public final Log log = LogFactory.getLog(this.getClass());
51  
52      private Pager usernamePager;
53      private UserAccessor userAccessor;
54  
55      public boolean isEmpty() {
56          Pager usernamePager = getUsernamePager();
57          if (usernamePager != null) {
58              usernamePager.nextPage();
59          }
60  
61          return getUsernamePager().isEmpty();
62      }
63  
64      public Iterator iterator() {
65          Pager usernamePager = getUsernamePager();
66          if (usernamePager != null) {
67              LazyLoadingUserByUsernamePagerIterator iterator = new LazyLoadingUserByUsernamePagerIterator();
68  
69              iterator.setUserAccessor(getUserAccessor());
70              iterator.setUsernamePagerIterator(getUsernamePager().iterator());
71                          
72              return iterator;
73          }
74  
75          return new ArrayList().iterator();
76      }
77  
78      public List getCurrentPage() {
79          List results = new ArrayList();
80          Pager usernamePager = getUsernamePager();
81          // must check for null here! (SUSR-54)
82          if ( usernamePager != null ) {
83              List usernames = usernamePager.getCurrentPage();
84              if (usernames!=null) {
85                  for (int i=0; i<usernames.size(); i++) {
86                      String username = (String)usernames.get(i);
87                      User user = getUserAccessor().getUser(username);
88                      results.add(user);
89                  }
90              }
91              else {
92                  log.debug("usernamePager.getCurrentPage() returned null. This may just be an empty group.");
93              }
94          }
95          else {
96              LogUtil.warnWithRemoteUserInfo(log, "usernamePager was null. Either was an empty group or an API issue. Look in debug logging for the groupname to look it up and be sure.");
97          }
98          return results;
99      }
100 
101     public void nextPage() {
102         Pager usernamePager = getUsernamePager();
103         if (usernamePager != null) {
104             usernamePager.nextPage();
105         }
106     }
107 
108     public boolean onLastPage() {
109         Pager usernamePager = getUsernamePager();
110         if (usernamePager != null) {
111             return usernamePager.onLastPage();
112         }
113 
114         return true;
115     }
116 
117     public void skipTo(int i) throws PagerException {
118         Pager usernamePager = getUsernamePager();
119         if (usernamePager != null) {
120             usernamePager.skipTo(i);
121         }
122     }
123 
124     public int getIndex() {

131     }
132 
133     public int getIndexOfFirstItemInCurrentPage() {
134         Pager usernamePager = getUsernamePager();
135         if (usernamePager != null) {
136             return usernamePager.getIndexOfFirstItemInCurrentPage();
137         }
138 
139         return 0;
140     }
141 
142     public Pager getUsernamePager() {
143         return usernamePager;
144     }

147         this.usernamePager = usernamePager;
148     }

151         return userAccessor;
152     }
