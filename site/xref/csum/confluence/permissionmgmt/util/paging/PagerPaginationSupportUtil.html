67      public static List toList(PagerPaginationSupport pps) {
68          //TODO: it is awful than PagerUtils.toList(pps.getItems()) doesn't work. submit a bug. this does way too many queries            
69  
70          log.debug("toList() called");
71          List result = null;
72          if (pps!=null) {
73              //TODO: may be being paranoid by using Map here. consider not using if not needed
74              // using map hopefully to avoid any chance of adding the same items more than once
75              Map itemsAsKeysMap = new HashMap();
76              int originalPageStartIndex = pps.getStartIndex();
77  
78              // start with first page
79              pps.setStartIndex(0);
80              log.debug("converting PagerPaginationSupport to List. at index=0");
81              addAllAsKeys(pps.getPage(), itemsAsKeysMap);
82  
83              // get other pages
84              int[] startIndexes = pps.getNextStartIndexes();
85              if (startIndexes!=null && startIndexes.length > 0) {
86                  for (int i=0; i<startIndexes.length; i++) {
87                      int nextStartIndex = startIndexes[i];
88                      log.debug("converting PagerPaginationSupport to List. at index=" + nextStartIndex);
89                      pps.setStartIndex(nextStartIndex);
90                      addAllAsKeys(pps.getPage(), itemsAsKeysMap);
91                  }
92              }
93  
94              result = new ArrayList(itemsAsKeysMap.keySet());;
95              if ( result.size() != pps.getTotal()) {
96                  LogUtil.warnWithRemoteUserInfo(log, "Got incorrect number of items in toList()! Expected " + pps.getTotal() + " but got " + result.size());
97              }
98              else {
99                  log.debug("Got " + result.size() + " items from pps. This was same as pps.getTotal() so hopefully all is ok.");                            
100             }
101         }
102         return result;
103     }

105     public static List getRanges(PagerPaginationSupport pps, int roughNumberOfRanges) {
106         log.debug("getRanges() called. roughNumberOfRanges=" + roughNumberOfRanges);
107         List ranges = null;
108         int numBeforeAndAfter = roughNumberOfRanges / 2;
109         if (pps!=null && pps.getTotal() > pps.getCountOnEachPage()) {
110             int[] startIndexes = pps.getNextStartIndexes();
111             if (startIndexes!=null && startIndexes.length > 0) {
112                 ranges = new ArrayList();
113 
114                 // ranges before current
115                 int currStartIndex = pps.getStartIndex();
116                 int lastCurrStartIndex = currStartIndex;
117                 for (int i=0; i<numBeforeAndAfter; i++) {
118                     currStartIndex = currStartIndex - pps.getCountOnEachPage();
119                     if ( currStartIndex < 0 && lastCurrStartIndex != 0) {
120                         currStartIndex = 0;
121                     }
122 
123                     if ( currStartIndex >= 0 ) {
124                         Range range = new Range();
125                         range.setRecordNum(currStartIndex + 1);
126                         range.setText("" + (currStartIndex + 1) + "-" + lastCurrStartIndex);
127                         ranges.add(range);
128                     }
129 
130                     lastCurrStartIndex = currStartIndex;
131                 }
132 
133                 // current
134                 ranges.add(getCurrentRange(pps));
135 
136                 // ranges after current
137                 int currEndIndex = pps.getStartIndex() + pps.getCountOnEachPage();
138                 int lastCurrEndIndex = currEndIndex;
139                 int maxIndex = pps.getTotal() - 1;
140                 for (int i=0; i<numBeforeAndAfter; i++) {
141                     currEndIndex = currEndIndex + pps.getCountOnEachPage();
142                     if ( currEndIndex > maxIndex && lastCurrEndIndex != maxIndex) {
143                         currEndIndex = maxIndex;
144                     }
145 
146                     if ( currEndIndex <= maxIndex ) {
147                         Range range = new Range();
148                         range.setRecordNum(lastCurrEndIndex + 1);
149                         range.setText("" + (lastCurrEndIndex + 1) + "-" + currEndIndex);
150                         ranges.add(range);
151                     }
152 
153                     lastCurrEndIndex = currEndIndex;
154                 }
155             }
156             else {
157                 log.debug("nextStartIndexes was null");
158             }
159         }
160         else {
161             log.debug("safelyMoveToPageStartIndexClosestToIndex() shouldn't really be called with null. programming error");
162         }
163 
164         return ranges;
165     }

167     private static Range getCurrentRange(PagerPaginationSupport pps) {
168         int lastRecordNumOfPage = pps.getStartIndex() + pps.getCountOnEachPage();
169         if (lastRecordNumOfPage > pps.getTotal()) {
170             lastRecordNumOfPage = pps.getTotal();
171         }
172         Range range = new Range();
173         range.setRecordNum(pps.getStartIndex() + 1);
174         range.setText("" + (pps.getStartIndex() + 1) + "-" + lastRecordNumOfPage);
175         return range;
176     }

200     public static void safelyMoveToOldStartIndex(int startIndex, PagerPaginationSupport pps) {
201         log.debug("safelyMoveToOldStartIndex() called. startIndex=" + startIndex);
202         if (pps!=null) {
203             int closestStartIndex = startIndex;
204             int maxIndex = pps.getTotal() - 1;
205             if (closestStartIndex > maxIndex) {
206                 closestStartIndex = maxIndex;
207             }
208             else if (closestStartIndex < 0) {
209                 closestStartIndex = 0;
210             }
211 
212             pps.setStartIndex(closestStartIndex);
213         }
214         else {
215             log.debug("safelyMoveToOldStartIndex() shouldn't really be called with null. programming error");
216         }
217     }

223     public static void safelyMoveToPageStartIndexClosestToIndex(int startIndex, PagerPaginationSupport pps) {
224         log.debug("safelyMoveToPageStartIndexClosestToIndex() called. startIndex=" + startIndex);
225         if (pps!=null) {
226             int[] startIndexes = pps.getNextStartIndexes();
227             if (startIndexes!=null && startIndexes.length > 0) {
228                 // 0 is a valid start index not included in nextStartIndexes
229                 int closestStartIndex = 0;
230                 for (int i=0; i<startIndexes.length; i++) {
231                     int nextStartIndex = startIndexes[i];
232                     int oldAbsoluteDifference = Math.abs(startIndex - closestStartIndex);
233                     int absoluteDifference = Math.abs(startIndex - nextStartIndex);
234 
235                     //log.debug("is startIndex=" + startIndex + " closer to " + closestStartIndex + " or " + nextStartIndex + " ?");
236 
237                     if ( absoluteDifference < oldAbsoluteDifference ) {
238                         //log.debug("now using " + nextStartIndex + " as closest index");
239                         closestStartIndex = nextStartIndex;
240                     }
241                 }
242 
243                 pps.setStartIndex(closestStartIndex);
244             }
245             else {
246                 log.debug("nextStartIndexes was null");
247             }
248         }
249         else {
250             log.debug("safelyMoveToPageStartIndexClosestToIndex() shouldn't really be called with null. programming error");
251         }

363     private static String intArrayToString(int[] ints) {
364         StringBuffer sb = new StringBuffer();
365         if (ints!=null) {
366             for (int i=0; i<ints.length; i++) {
367                 if (i!=0) {
368                     sb.append(", ");
369                 }
370                 sb.append( ints[i] );
371             }
372         }
373         else {
374             sb.append("null");
375         }
376         return sb.toString();
377     }
