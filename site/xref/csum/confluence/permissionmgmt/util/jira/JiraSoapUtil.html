30  package csum.confluence.permissionmgmt.util.jira;
31  
32  import csum.confluence.permissionmgmt.util.jira.JiraServiceAuthenticationContext;
33  import csum.confluence.permissionmgmt.util.logging.LogUtil;
34  import csum.confluence.permissionmgmt.service.exception.ServiceAuthenticationException;
35  import csum.confluence.permissionmgmt.service.vo.ServiceContext;
36  import csum.confluence.permissionmgmt.soap.jira.JiraSoapServiceServiceLocator;
37  import csum.confluence.permissionmgmt.soap.jira.JiraSoapService;
38  import csum.confluence.permissionmgmt.config.CustomPermissionConfigurable;
39  import org.apache.commons.logging.Log;
40  import org.apache.commons.logging.LogFactory;
41  
42  import javax.xml.rpc.ServiceException;
43  
44  /***
45   * @author Gary S. Weaver
46   */
47  public class JiraSoapUtil {
48  
49      private static Log log = LogFactory.getLog(JiraSoapUtil.class);
50  
51      public static JiraServiceAuthenticationContext login(ServiceContext serviceContext) throws ServiceAuthenticationException {
52          JiraServiceAuthenticationContext jiraServiceAuthenticationContext = new JiraServiceAuthenticationContext();
53          try {
54              JiraSoapServiceServiceLocator jiraSoapServiceGetter = new JiraSoapServiceServiceLocator();
55              CustomPermissionConfigurable config = serviceContext.getCustomPermissionConfigurable();
56              jiraSoapServiceGetter.setJirasoapserviceV2EndpointAddress(config.getJiraSoapUrl());
57              JiraSoapService jiraSoapService = jiraSoapServiceGetter.getJirasoapserviceV2();
58              jiraServiceAuthenticationContext.setJiraSoapService( jiraSoapService );
59              jiraServiceAuthenticationContext.setToken( jiraSoapService.login(config.getJiraSoapUsername(), config.getJiraSoapPassword()) );
60          }
61          catch (Throwable t) {
62              LogUtil.errorWithRemoteUserInfo(log, "Failed Jira SOAP Authentication!", t);
63              throw new ServiceAuthenticationException("Failed Jira SOAP Authentication!", t);
64          }
65  
66          return jiraServiceAuthenticationContext;
67      }
68  
69      public static void logout(JiraServiceAuthenticationContext jiraServiceAuthenticationContext) throws ServiceAuthenticationException {
70          try {
71              jiraServiceAuthenticationContext.getJiraSoapService().logout(jiraServiceAuthenticationContext.getToken());
72          }
73          catch (Throwable t) {
74              LogUtil.errorWithRemoteUserInfo(log, "Failed Jira SOAP Authentication!", t);
75              throw new ServiceAuthenticationException("Failed Jira SOAP Authentication!", t);
76          }
77      }
78  }
