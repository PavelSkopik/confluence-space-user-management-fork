205     private String cleanParamValue(String param, String value) {
206         if (value!=null) {
207             value = StringUtil.clean(value.trim());
208             if (!StringUtil.isNullOrEmpty(value)) {
209                 try {
210                     value = URLDecoder.decode(value, "UTF-8");
211                 }
212                 catch (UnsupportedEncodingException e) {
213                     // NOTE: this should allow legal groupnames to pass so just log and ignore
214                     log.debug("request parameter '" + param + "' had cleaned up value '" + value + "' which could not be URL decoded", e);
215                 }
216                 catch (IllegalArgumentException e) {
217                     //TODO: write test for groupname sdfaa#$%#$%$%&
218                     // NOTE: this should allow legal groupnames to pass so just log and ignore
219                     log.debug("request parameter '" + param + "' had cleaned up value '" + value + "' which could not be URL decoded", e);
220                 }
221             }
222         }
223         return value;
224     }

333     private void handlePaging(Map paramMap) {
334         setPagerAction(getUrlDecodedCleanedTrimmedParameterValue( paramMap, "pagerAction"));
335         String pagerAction = getPagerAction();
336         if ("nextPageGroups".equals(pagerAction)) {
337             PagerPaginationSupport groups = getGroups();
338             if (hasNext(groups)) {
339                 next(groups);
340             }
341         }
342         else if ("prevPageGroups".equals(pagerAction)) {
343             PagerPaginationSupport groups = getGroups();
344             if (hasPrev(groups)) {
345                 prev(groups);
346             }
347         }
348         else if ("skipToGroup".equals(pagerAction)) {
349             Integer recordNum = getRecordNum(paramMap);
350             if (recordNum!=null) {
351                 // record num is one-based, not zero-based
352                 PagerPaginationSupportUtil.safelyMoveToOldStartIndex(new Integer(recordNum.intValue() - 1), getGroups());
353             }
354         }
355 
356         if ("nextPageUsers".equals(pagerAction)) {
357             PagerPaginationSupport users = getUsers();
358             if (hasNext(users)) {
359                 next(users);
360             }
361         }
362         else if ("prevPageUsers".equals(pagerAction)) {
363             PagerPaginationSupport users = getUsers();
364             if (hasPrev(users)) {
365                 prev(users);
366             }
367         }
368         else if ("skipToUser".equals(pagerAction)) {
369             Integer recordNum = getRecordNum(paramMap);
370             if (recordNum!=null) {
371                 // record num is one-based, not zero-based
372                 PagerPaginationSupportUtil.safelyMoveToOldStartIndex(new Integer(recordNum.intValue() - 1), getUsers());
373             }
374         }
375 
376         if ("nextPageSearchResultUsers".equals(pagerAction)) {
377             PagerPaginationSupport users = getSearchResultUsers();
378             if (hasNext(users)) {
379                 next(users);
380             }
381         }
382         else if ("prevPageSearchResultUsers".equals(pagerAction)) {
383             PagerPaginationSupport users = getSearchResultUsers();
384             if (hasPrev(users)) {
385                 prev(users);
386             }
387         }
388         else if ("skipToSearchResultUser".equals(pagerAction)) {
389             Integer recordNum = getRecordNum(paramMap);
390             if (recordNum!=null) {
391                 // record num is one-based, not zero-based
392                 PagerPaginationSupportUtil.safelyMoveToOldStartIndex(new Integer(recordNum.intValue() - 1), getSearchResultUsers());
393             }
394         }
395     }

473     // this helps get around bug in atlassian's API CSP-10371/CONF-9035 to be fixed in confluence 2.6
474     private void handleRefreshBugFirstRequest(String result) {
475         HttpServletRequest req = ServletActionContext.getRequest();
476         // note: wrapping with TreeMap so it will sort params by name, otherwise it makes inconsistent URL which looks
477         //       hackish.
478         Map paramMap = new TreeMap(ServletActionContext.getRequest().getParameterMap());
479         String[] redirect = (String[])paramMap.get(REDIRECT_PARAMNAME);
480         if (redirect!=null) {
481             HttpServletResponse resp = ServletActionContext.getResponse();
482 
483             // Redirect HTTP POST as well as HTTP GET. Remove REDIRECT_PARAMNAME from params.
484             StringBuffer params = new StringBuffer();
485             params.append("?");
486             Iterator iter = paramMap.keySet().iterator();
487             String paramConcat = "";
488             while (iter.hasNext()) {
489                 String param = (String)iter.next();
490 
491                 // remove any params that could either do an additional redirect or additional action, except for groups
492                 // because will need groups to know what cache to refresh on second request
493                 if (!REDIRECT_PARAMNAME.equalsIgnoreCase(param) &&
494                         !ADMIN_ACTION_PARAMNAME.equalsIgnoreCase(param) &&
495                         !USERS_PARAMNAME.equalsIgnoreCase(param)
496                         ) {
497                     params.append(paramConcat);
498                     params.append(bestAttemptUTF8Encode(param));
499                     params.append("=");
500                     String[] value = (String[])paramMap.get(param);
501                     String valueConcat = "";
502                     if (value!=null) {
503                         params.append(valueConcat);
504                         for (int i=0;i<value.length;i++) {
505                             params.append(bestAttemptUTF8Encode(value[i]));
506                         }
507                         valueConcat = ",";
508                     }
509                     paramConcat = "&";
510                 }
511             }
512 
513             // store result, etc in session to be removed after redirect
514             storeInPluginCache("result", result);
515             storeInPluginCache("actionmessages", this.getActionMessages());
516             storeInPluginCache("actionerrors", this.getActionErrors());
517             storeInPluginCache("fielderrors", this.getFieldErrors());
518 
519             // add a param that tells action to refresh only cache associated with currently selected group in next req
520             params.append("&" + REFRESH_BUG_SECOND_REQUEST_PARAMNAME);
521 
522             String url = req.getRequestURI();
523             int indexOfQuery = url.indexOf('?');
524             if (indexOfQuery != -1) {
525                 // chop off query
526                 url = url.substring(0, indexOfQuery);
527             }
528             // readd query with any post params
529             url = url + params.toString();
530 
531             log.debug("Workaround for bug CSP-10371. Redirecting to: " + url);
532             try {
533                 resp.sendRedirect( url );
534             }
535             catch (IOException e) {
536                 LogUtil.errorWithRemoteUserInfo(log, "Redirect failed!", e);
537             }
538         }
539     }

541     // this helps get around bug in atlassian's API CSP-10371/CONF-9035 to be fixed in confluence 2.6
542     private String handleRefreshBugSecondRequest(Map paramMap) {
543 
544         String result = null;
545 
546         if (this.getRawParameterValue(paramMap, REFRESH_BUG_SECOND_REQUEST_PARAMNAME) != null) {
547 
548             // RECOVER STATE
549 
550             result = (String)getFromPluginCache("result");
551             setActionMessages((Collection)getFromPluginCache("actionmessages"));
552             setActionErrors((Collection)getFromPluginCache("actionerrors"));
553             setFieldErrors((Map)getFromPluginCache("fielderrors"));
554 
555             // CLEAR STATE
556 
557             removeFromPluginCache("result");
558             removeFromPluginCache("actionmessages");
559             removeFromPluginCache("actionerrors");
560             removeFromPluginCache("fielderrors");
561 
562             // CLEAR SELECTED CACHE
563 
564             this.clearGroupCache(getKey());
565             // this is why we have to leave groups param in the url in handleRedirect
566             List groups = getUrlDecodedCleanedTrimmedParameterValueList(paramMap, GROUPS_PARAMNAME);
567             if (groups!=null) { 
568                 this.clearSearchResultUserCache(getKey(), groups);
569                 doUserSearch();
570                 this.clearUserCache(getKey(), groups);
571             }
572         }
573 
574         return result;
575     }
