34  /***
35   * @author Gary S. Weaver
36   */
37  public class AdvancedUserQuery {
38  
39      private String lookupType = AdvancedUserQueryLookupType.USERNAME;
40      private String partialSearchTerm;
41      private String substringMatchType = AdvancedUserQuerySubstringMatchType.SUBSTRING_STARTS_WITH;
42  
43      public String toString() {
44          return "lookupType=" + lookupType + " partialSearchTerm=" + partialSearchTerm + " substringMatchType=" + substringMatchType;
45      }
46  
47      public String getLookupType() {
48          return lookupType;
49      }
50  
51      public void setLookupType(String lookupType) {
52          this.lookupType = lookupTypeInstance(lookupType);
53      }
54  
55      public String getPartialSearchTerm() {
56          return partialSearchTerm;
57      }
58  
59      public void setPartialSearchTerm(String partialSearchTerm) {
60          this.partialSearchTerm = partialSearchTerm;
61      }
62  
63      public String getSubstringMatchType() {
64          return substringMatchType;
65      }
66  
67      public void setSubstringMatchType(String substringMatchType) {
68          this.substringMatchType = substringMatchTypeInstance(substringMatchType);
69      }
70  
71      public boolean isDefined() {
72          boolean result = false;
73          if ( this.getLookupType() != null ||
74                  this.getSubstringMatchType() != null ||
75                  !ConfigUtil.isNullOrEmpty(this.getPartialSearchTerm())) {
76              result = true;
77          }
78          return result;
79      }
80  
81      // TODO: consider using enumeration instance instead
82      public String lookupTypeInstance(String s) {
83          String result = null;
84          if (s!=null) {
85              if (AdvancedUserQueryLookupType.USERNAME.equalsIgnoreCase(s) ) {
86                  result = AdvancedUserQueryLookupType.USERNAME;
87              }
88              else if (AdvancedUserQueryLookupType.USER_FULL_NAME.equalsIgnoreCase(s) ) {
89                  result = AdvancedUserQueryLookupType.USER_FULL_NAME;
90              }
91              else if (AdvancedUserQueryLookupType.USER_EMAIL.equalsIgnoreCase(s) ) {
92                  result = AdvancedUserQueryLookupType.USER_EMAIL;
93              }
94          }
95          return result;
96      }
97  
98      // note: atlassian-user is really picky about doing == instead of .equals to check the TermQuery constant, so we do this
99      public String substringMatchTypeInstance(String s) {
100         String result = null;
101         if (s!=null) {
102             if (AdvancedUserQuerySubstringMatchType.SUBSTRING_CONTAINS.equalsIgnoreCase(s) ) {
103                 result = AdvancedUserQuerySubstringMatchType.SUBSTRING_CONTAINS;
104             }
105             else if (AdvancedUserQuerySubstringMatchType.SUBSTRING_ENDS_WITH.equalsIgnoreCase(s) ) {
106                 result = AdvancedUserQuerySubstringMatchType.SUBSTRING_ENDS_WITH;
107             }
108             else if (AdvancedUserQuerySubstringMatchType.SUBSTRING_STARTS_WITH.equalsIgnoreCase(s) ) {
109                 result = AdvancedUserQuerySubstringMatchType.SUBSTRING_STARTS_WITH;
110             }
111         }
112         return result;
113     }
114 
115     public boolean isUsernameSearchDefined() {
116         boolean result = false;
117         if (this.getLookupType()==AdvancedUserQueryLookupType.USERNAME) {
118             result = true;
119         }
120         return result;
121     }
122 
123     public boolean isFullnameSearchDefined() {
124         boolean result = false;
125         if (this.getLookupType()==AdvancedUserQueryLookupType.USER_FULL_NAME) {
126             result = true;
127         }
128         return result;
129     }
130 
131     public boolean isEmailSearchDefined() {
132         boolean result = false;
133         if (this.getLookupType()==AdvancedUserQueryLookupType.USER_EMAIL) {
134             result = true;
135         }
136         return result;
137     }
138 
139     public boolean isValid()
140     {
141         return isDefined();
142     }
