56      public void addUsersByUsernameToGroups(List userNames, List groupNames, ServiceContext context) throws UsersNotFoundException, AddException {
57          log.debug("addUsersByUsernameToGroupsByGroupname() called. " +
58                  "usernames=" + StringUtil.convertCollectionToCommaDelimitedString(userNames) +
59                  ", groupnames=" + StringUtil.convertCollectionToCommaDelimitedString(groupNames));
60          CustomPermissionConfiguration config = getCustomPermissionConfiguration();
61  
62          List usersNotFound = new ArrayList();
63          Map userIdToGroupNameMapForMembershipAdditionProblems = new TreeMap();
64          // using map to get only unique groups. using treemap to keep groupnames in order
65          Map groupsNotFoundMap = new TreeMap();
66  
67          boolean isLDAPPresent = config.getLdapAuthUsed().equals(CustomPermissionConfigConstants.YES) ? true : false;
68  
69          //Associate selected user-groups to all users.
70          for (Iterator itr = userNames.iterator(); itr.hasNext();) {
71              //First check if given user is present or not
72              String userid = (String) itr.next();
73              User user = userAccessor.getUser(userid);
74              if (user == null) {
75                  //create an user
76                  //userid doesn't exists, if LDAP present then we will create User if it exists in LDAP.
77                  if (isLDAPPresent) {
78                      //create an user.
79  
80                      // TODO: the option to create users if they don't exist using LDAP info should be in config
81  
82                      // TODO: consider adding option and ability to create users if they don't exist, even if LDAP not used
83  
84                      user = createConfUser(userid, isLDAPPresent);
85                  }
86  
87                  //if user details not found in LDAP too, then retun userid in errorids
88                  if (user == null) {
89  
90                      //for some reason we are unable to create user.
91                      //add it to our notCreatedUser List.
92                      usersNotFound.add(userid);
93  
94                      continue;
95  
96                  } else {
97                      //Add this user to default group confluence-users
98                      addMembershipPassivelyAndTrackingErrors(ServiceConstants.CONFLUENCE_USERS_GROUP_NAME, user, groupsNotFoundMap, usersNotFound, userIdToGroupNameMapForMembershipAdditionProblems);
99                  }
100             }
101 
102             //If user exists then associate him/her to all specified groups
103             if (user != null) {
104                 //Associate this user to all selected user-groups
105                 for (Iterator iterator = groupNames.iterator(); iterator.hasNext();) {
106                     String groupName = (String) iterator.next();
107                     addMembershipPassivelyAndTrackingErrors(groupName, user, groupsNotFoundMap, usersNotFound, userIdToGroupNameMapForMembershipAdditionProblems);
108                 }
109             }
110         }
111 
112         // If we failed, throw exception
113         List groupsNotFound = new ArrayList(groupsNotFoundMap.keySet());
114         if (usersNotFound.size() > 0 && groupsNotFound.size() == 0 && userIdToGroupNameMapForMembershipAdditionProblems.size() == 0) {
115             // a less critical error that we don't want to log the same way
116             throw new UsersNotFoundException(context.getText("manager.error.usersnotfound") + ": " +
117                     StringUtil.convertCollectionToCommaDelimitedString(usersNotFound) + ".");
118         }
119         else if (groupsNotFound.size() > 0 || userIdToGroupNameMapForMembershipAdditionProblems.size() > 0) {
120             throw new AddException(getRemoveUsersByUsernameFromGroupsErrorMessage(usersNotFound, groupsNotFound, userIdToGroupNameMapForMembershipAdditionProblems, context));
121         }
122     }

124     //This method will be used to create an user when Confluence is used for Managing Wiki Users
125     private User createConfUser(String creationUserName, boolean isLDAPAvailable) {
126         log.debug("createConfUser() called. creationUserName=" + creationUserName + " isLDAPAvailable" + isLDAPAvailable);
127         User vUser = null;
128         LDAPUser lUser = null;
129 
130         try {
131             //if LDAP Lookup is available, get information from there.
132             if (isLDAPAvailable) {
133                 //log.debug("LDAP Lookup available");
134                 //Get user details from LDAP.
135                 lUser = getLDAPUser(creationUserName);
136 
137                 if (lUser != null) {
138                     vUser = userAccessor.addUser(creationUserName, creationUserName, lUser.getEmail(), lUser.getFullName());
139                 }
140             }
141         } catch (Exception e) {
142             LogUtil.errorWithRemoteUserInfo(log, "Error creating confluence user " + creationUserName, e);
143         }
144 
145         return vUser;
146     }

148     private void addMembershipPassivelyAndTrackingErrors(String groupName, User user, Map groupsNotFoundMap, List usersNotFound, Map userIdToGroupNameMapForMembershipAdditionProblems) {
149         if (groupsNotFoundMap.get(groupName) == null && user != null) {
150             try {
151                 Group group = userAccessor.getGroup(groupName);
152                 if (group == null) {
153 
154                     String lowercaseGroupName = groupName.toLowerCase();
155                     if (lowercaseGroupName.equals(groupName)) {
156                         // group name was already lowercase
157                         LogUtil.warnWithRemoteUserInfo(log, "Failed adding " + user.getName() + " to " + groupName + ". Group didn't exist");
158                         groupsNotFoundMap.put("" + groupName, "");
159                         return;
160                     } else {
161                         // There is a bug where groupname must be lowercase but we handle both
162                         // scenarios just in case it gets fixed: http://jira.atlassian.com/browse/CONF-9224
163 
164                         log.debug("No group exists for groupname " + groupName + " so will try lowercase groupName");
165                         group = userAccessor.getGroup(lowercaseGroupName);
166                         if (group == null) {
167                             LogUtil.warnWithRemoteUserInfo(log, "Failed adding " + user.getName() + " to " + groupName + ". Group didn't exist (tried regular case and lowercase groupname)");
168                             // make sure to use regular case here since it is checked against in calling method
169                             groupsNotFoundMap.put("" + groupName, "");
170                         }
171                     }
172                 }
173 
174                 if (group != null) {
175                     log.debug("Adding " + user.getName() + " to group " + groupName);
176                     userAccessor.addMembership(group, user);
177                 }
178             }
179             catch (Throwable t) {
180                 LogUtil.errorWithRemoteUserInfo(log, "Failed adding " + user.getName() + " to " + groupName);
181                 // using "" + to guard against nulls
182                 userIdToGroupNameMapForMembershipAdditionProblems.put("" + user.getName(), "" + groupName);
183             }
184         }
185     }

187     public void removeUsersByUsernameFromGroups(List userNames, List groupNames, ServiceContext context) throws UsersNotFoundException, RemoveException {
188         log.debug("removeUsersByUsernameFromGroups() called. " +
189                 "usernames=" + StringUtil.convertCollectionToCommaDelimitedString(userNames) +
190                 ", groupnames=" + StringUtil.convertCollectionToCommaDelimitedString(groupNames));
191 
192         List usersNotFound = new ArrayList();
193         Map userIdToGroupNameMapForMembershipRemovalProblems = new TreeMap();
194         // using map to get only unique groups. using treemap to keep groupnames in order
195         Map groupsNotFoundMap = new TreeMap();
196 
197         for (Iterator itr = userNames.iterator(); itr.hasNext();) {
198             String userid = (String) itr.next();
199             User user = userAccessor.getUser(userid);
200             if (user != null) {
201                 for (Iterator iterator = groupNames.iterator(); iterator.hasNext();) {
202                     String groupName = (String) iterator.next();
203                     removeMembershipPassivelyAndTrackingErrors(groupName, user, groupsNotFoundMap, usersNotFound, userIdToGroupNameMapForMembershipRemovalProblems);
204                 }
205             } else {
206                 usersNotFound.add(userid);
207             }
208         }
209

212         if (usersNotFound.size() > 0 && groupsNotFound.size() == 0 && userIdToGroupNameMapForMembershipRemovalProblems.size() == 0) {
213             // a less critical error that we don't want to log the same way
214             throw new UsersNotFoundException(context.getText("manager.error.usersnotfound") + ": " +
215                     StringUtil.convertCollectionToCommaDelimitedString(usersNotFound) + ".");
216         }
217         else if (groupsNotFound.size() > 0 || userIdToGroupNameMapForMembershipRemovalProblems.size() > 0) {
218             throw new RemoveException(getRemoveUsersByUsernameFromGroupsErrorMessage(usersNotFound, groupsNotFound, userIdToGroupNameMapForMembershipRemovalProblems, context));
