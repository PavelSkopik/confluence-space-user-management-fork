64      public void addUsersByUsernameToGroups(List userNames, List groupNames, ServiceContext context) throws UsersNotFoundException, AddException, ServiceAuthenticationException {
65          log.debug("addUsersByUsernameToGroupsByGroupname() called. " +
66                  "usernames=" + StringUtil.convertCollectionToCommaDelimitedString(userNames) +
67                  ", groupnames=" + StringUtil.convertCollectionToCommaDelimitedString(groupNames));
68  
69          JiraServiceAuthenticationContext authContext = null;
70          List usersNotFound = new ArrayList();
71          Map userIdToGroupNameMapForMembershipAdditionProblems = new TreeMap();
72          // using map to get only unique groups. using treemap to keep groupnames in order
73          Map groupsNotFoundMap = new TreeMap();
74  
75          try {
76              authContext = JiraSoapUtil.login(context);
77              JiraSoapService jiraSoapService = authContext.getJiraSoapService();
78              String token = authContext.getToken();
79  
80              CustomPermissionConfiguration config = getCustomPermissionConfiguration();
81  
82              boolean isLDAPPresent = config.getLdapAuthUsed().equals(CustomPermissionConfigConstants.YES) ? true : false;
83  
84              //Associate selected user-groups to all users.
85              for (Iterator itr = userNames.iterator(); itr.hasNext();) {
86                  //First check if given user is present or not
87                  String userid = (String) itr.next();
88  
89                  if (!usersNotFound.contains(userid)) {
90                      RemoteUser remoteUser = jiraSoapService.getUser(token, userid);
91                      if (remoteUser == null) {
92                          //userid doesn't exist, but if LDAP present then we will create user if it exists in LDAP.
93                          if (isLDAPPresent) {
94  							System.out.println("Getting user from LDAP ...");
95                              remoteUser = createJiraUser(token, jiraSoapService, userid, isLDAPPresent);
96                          }
97  
98                          //if user details not found in LDAP too, then return userid in errorids
99                          if (remoteUser == null) {
100 
101                             //for some reason we are unable to create user.
102                             //add it to our notCreatedUser List.
103                             usersNotFound.add(userid);
104 
105                             continue;
106 
107                         } else {
108                             addMembershipToJiraPassivelyAndTrackingErrors(jiraSoapService, token, ServiceConstants.CONFLUENCE_USERS_GROUP_NAME, remoteUser, groupsNotFoundMap, usersNotFound, userIdToGroupNameMapForMembershipAdditionProblems);
109                             addMembershipToJiraPassivelyAndTrackingErrors(jiraSoapService, token, ServiceConstants.JIRA_USERS_GROUP_NAME, remoteUser, groupsNotFoundMap, usersNotFound, userIdToGroupNameMapForMembershipAdditionProblems);
110                         }
111                     }
112 
113                     //If user exists then associate him/her to all selected User
114                     if (remoteUser != null) {
115                         //Associate this user to all selected user-groups
116                         for (Iterator iterator = groupNames.iterator(); iterator.hasNext();) {
117                             String groupName = (String) iterator.next();
118                             addMembershipToJiraPassivelyAndTrackingErrors(jiraSoapService, token, groupName, remoteUser, groupsNotFoundMap, usersNotFound, userIdToGroupNameMapForMembershipAdditionProblems);
119                         }
120                     }
121                 }
122             }
123         }
124         catch (Throwable e) {
125             LogUtil.errorWithRemoteUserInfo(log, "Failed adding users to groups!", e);
126             throw new AddException(e.getMessage(), e);
127         }
128         finally {
129             if (authContext != null) {
130                 try {
131                     JiraSoapUtil.logout(authContext);
132                 }
133                 catch (Throwable t) {
134                     LogUtil.errorWithRemoteUserInfo(log, "Error in Jira logout", t);
135                 }
136             }
137         }
138 
139         // If we failed, throw exception
140         List groupsNotFound = new ArrayList(groupsNotFoundMap.keySet());
141         if (usersNotFound.size() > 0 && groupsNotFound.size() == 0 && userIdToGroupNameMapForMembershipAdditionProblems.size() == 0) {
142             // a less critical error that we don't want to log the same way
143             throw new UsersNotFoundException(context.getText("manager.error.usersnotfound") + ": " +
144                     StringUtil.convertCollectionToCommaDelimitedString(usersNotFound) + ".");
145         }
146         else if (groupsNotFound.size() > 0 || userIdToGroupNameMapForMembershipAdditionProblems.size() > 0) {
147             throw new AddException(getRemoveUsersByUsernameFromGroupsErrorMessage(usersNotFound, groupsNotFound, userIdToGroupNameMapForMembershipAdditionProblems, context));
148         }
149     }

151     //This method will be used to create an user when Confluence is used for Managing Wiki Users
152     private RemoteUser createJiraUser(String token, JiraSoapService jiraSoapService, String creationUserName, boolean isLDAPAvailable) {
153         log.debug("createConfUser() called. creationUserName=" + creationUserName + " isLDAPAvailable" + isLDAPAvailable);
154         RemoteUser vUser = null;
155         LDAPUser lUser = null;
156 
157         try {
158             //if LDAP Lookup is available, get information from there.
159             if (isLDAPAvailable) {
160                 //log.debug("LDAP Lookup available");
161                 //Get user details from LDAP.
162                 lUser = getLDAPUser(creationUserName);
163 
164                 if (lUser != null) {
165                     //createUser(String token, String username, String password, String fullName, String email)
166                     vUser = jiraSoapService.createUser(token, creationUserName, creationUserName, lUser.getFullName(), lUser.getEmail() );
167                 }
168             }
169         } catch (Throwable e) {
170             LogUtil.errorWithRemoteUserInfo(log, "Error creating Jira user", e);
171         }
172 
173         return vUser;
174     }

176     private void addMembershipToJiraPassivelyAndTrackingErrors(JiraSoapService jiraSoapService, String token, String groupName, RemoteUser remoteUser, Map groupsNotFoundMap, List usersNotFound, Map userIdToGroupNameMapForMembershipAdditionProblems) {
177         if (groupsNotFoundMap.get(groupName) == null && remoteUser != null) {
178             try {
179                 RemoteGroup remoteGroup = jiraSoapService.getGroup(token, groupName);
180                 if (remoteGroup == null) {
181 
182                     String lowercaseGroupName = groupName.toLowerCase();
183                     if (lowercaseGroupName.equals(groupName)) {
184                         // group name was already lowercase
185                         LogUtil.warnWithRemoteUserInfo(log, "Failed adding " + remoteUser.getName() + " to " + groupName + " in Jira. Jira group didn't exist");
186                         groupsNotFoundMap.put("" + groupName, "");
187                         return;
188                     } else {
189                         // There is a bug where groupname must be lowercase but we handle both
190                         // scenarios just in case it gets fixed: http://jira.atlassian.com/browse/CONF-9224
191 
192                         log.debug("No Jira group exists for groupname " + groupName + " so will try lowercase groupName");
193                         remoteGroup = jiraSoapService.getGroup(token, lowercaseGroupName);
194                         if (remoteGroup == null) {
195                             LogUtil.warnWithRemoteUserInfo(log, "Failed adding " + remoteUser.getName() + " to " + groupName + " in Jira. Jira group didn't exist (tried regular case and lowercase groupname)");
196                             // make sure to use regular case here since it is checked against in calling method
197                             groupsNotFoundMap.put("" + groupName, "");
198                         }
199                     }
200                 }
201 
202                 if (remoteGroup != null) {
203                     log.debug("Adding " + remoteUser.getName() + " to Jira group " + groupName);
204                     jiraSoapService.addUserToGroup(token, remoteGroup, remoteUser);
205                 }
206             }
207             catch (Throwable t) {
208                 LogUtil.errorWithRemoteUserInfo(log, "Failed adding " + remoteUser.getName() + " to " + groupName);
209                 // using "" + to guard against nulls
210                 userIdToGroupNameMapForMembershipAdditionProblems.put("" + remoteUser.getName(), "" + groupName);
211             }
212         }
213     }

215     public void removeUsersByUsernameFromGroups(List userNames, List groupNames, ServiceContext context) throws UsersNotFoundException, RemoveException, ServiceAuthenticationException {
216         log.debug("removeUsersByUsernameFromGroups() called. " +
217                 "usernames=" + StringUtil.convertCollectionToCommaDelimitedString(userNames) +
218                 ", groupnames=" + StringUtil.convertCollectionToCommaDelimitedString(groupNames));
219         JiraServiceAuthenticationContext authContext = null;
220 
221         List usersNotFound = new ArrayList();
222         Map userIdToGroupNameMapForMembershipRemovalProblems = new TreeMap();
223         // using map to get only unique groups. using treemap to keep groupnames in order
224         Map groupsNotFoundMap = new TreeMap();
225 
226         try {
227             authContext = JiraSoapUtil.login(context);
228             JiraSoapService jiraSoapService = authContext.getJiraSoapService();
229             String token = authContext.getToken();
230 
231             for (Iterator itr = userNames.iterator(); itr.hasNext();) {
232                 String userid = (String) itr.next();
233 
234                 RemoteUser remoteUser = authContext.getJiraSoapService().getUser(token, userid);
235                 if (remoteUser != null) {
236                     for (Iterator iterator = groupNames.iterator(); iterator.hasNext();) {
237                         String groupName = (String) iterator.next();
238                         removeMembershipFromJiraPassivelyAndTrackingErrors(jiraSoapService, token, groupName, remoteUser, groupsNotFoundMap, usersNotFound, userIdToGroupNameMapForMembershipRemovalProblems);
239                     }
240                 } else {
241                     usersNotFound.add(userid);
242                 }
243             }
244         }
245         catch (Throwable e) {
246             LogUtil.errorWithRemoteUserInfo(log, "Failed removing users from groups!", e);
247             throw new RemoveException(e.getMessage(), e);
248         }
249         finally {
250             if (authContext != null) {
251                 try {
252                     JiraSoapUtil.logout(authContext);
253                 }

255                     LogUtil.errorWithRemoteUserInfo(log, "Error in Jira logout", t);

257             }
258         }
259

261         List groupsNotFound = new ArrayList(groupsNotFoundMap.keySet());
262         if (usersNotFound.size() > 0 && groupsNotFound.size() == 0 && userIdToGroupNameMapForMembershipRemovalProblems.size() == 0) {
263             // a less critical error that we don't want to log the same way
264             throw new UsersNotFoundException(context.getText("manager.error.usersnotfound") + ": " +
265                     StringUtil.convertCollectionToCommaDelimitedString(usersNotFound) + ".");
266         }
267         else if (groupsNotFound.size() > 0 || userIdToGroupNameMapForMembershipRemovalProblems.size() > 0) {
268             throw new RemoveException(getRemoveUsersByUsernameFromGroupsErrorMessage(usersNotFound, groupsNotFound, userIdToGroupNameMapForMembershipRemovalProblems, context));
