89      public void updateWith(CustomPermissionConfigurable config) {
90          setUserManagerLocation(config.getUserManagerLocation());
91          setMaxUserIDsLimit(config.getMaxUserIDsLimit());
92          setMaxGroupIDsLimit(config.getMaxGroupIDsLimit());
93          setLdapAuthUsed(config.getLdapAuthUsed());
94          setPluginDown(config.getPluginDown());
95          setDownTimeMessage(config.getDownTimeMessage());
96          setGroupActionsPermitted(config.getGroupActionsPermitted());
97          setNewGroupNameCreationPrefixPattern(config.getNewGroupNameCreationPrefixPattern());
98          setNewGroupNameCreationSuffixPattern(config.getNewGroupNameCreationSuffixPattern());
99          setUserSearchEnabled(config.getUserSearchEnabled());
100         setJiraSoapUrl(config.getJiraSoapUrl());
101         setJiraSoapUsername(config.getJiraSoapUsername());
102 
103         String jiraSoapPassword = config.getJiraSoapPassword();
104         if (jiraSoapPassword!=null) {
105             // only change password if it is set to a string (can be empty, but if null that indicates not to change)
106             setJiraSoapPassword(jiraSoapPassword);
107         }
108 
109         setProviderType(config.getProviderType());
110         setLdapUserIdAttribute(config.getLdapUserIdAttribute());
111         setLdapEmailAttribute(config.getLdapEmailAttribute());
112         setLdapFirstNameAttribute(config.getLdapFirstNameAttribute());
113         setLdapLastNameAttribute(config.getLdapLastNameAttribute());
114         setLdapProviderFullyQualifiedClassname(config.getLdapProviderFullyQualifiedClassname());
115         setUserFullNameFormat(config.getUserFullNameFormat());
116         setLdapConfigTestUsername(config.getLdapConfigTestUsername());
117         setLdapNarrowingFilterExpression(config.getLdapNarrowingFilterExpression());
118         setPersonalSpaceAllowed(config.getPersonalSpaceAllowed());
119 
120         // config has changed. clear ALL cache including indexes!!!
121         AbstractPagerPaginationSupportCachingSpaceAction.clearCacheIncludingIndexes();
122     }

134     public static CsumConfigValidationResponse validate(CustomPermissionConfigurable config, CustomPermissionConfigurable existingConfig, String remoteUser, ConfluenceActionSupport cas, boolean shouldTestExternalServices) {
135 
136         CsumConfigValidationResponse result = new CsumConfigValidationResponse();
137         result.setValid(true);
138         
139 		String userMgrLocation = config.getUserManagerLocation();
140 		boolean isUserManagerLocationSet = (userMgrLocation != null);
141         boolean userManagerLocationIsConfluence = (userMgrLocation != null) && (userMgrLocation.equals(CustomPermissionConfigConstants.DELEGATE_USER_MANAGER_LOCATION_CONFLUENCE_VALUE));
142 		boolean userManagerLocationIsJira = (userMgrLocation != null) && (userMgrLocation.equals(CustomPermissionConfigConstants.DELEGATE_USER_MANAGER_LOCATION_JIRA_VALUE));
143 
144 		//If userManagerLocation is not set as CONFLUENCE or JIRA, then it must be set to either value.
145 		if( !(userManagerLocationIsConfluence || userManagerLocationIsJira) )
146 		{
147             result.addFieldError("userManagerLocation", cas.getText("configure.error.usermanagerlocationnull"));
148             result.setValid(false);
149 		}
150 
151 		//Following information needs to be check only if Wiki User Management is delegated to Jira
152 		if(isUserManagerLocationSet && userManagerLocationIsJira)
153 		{
154 			if (ConfigUtil.isNullOrEmpty(config.getJiraSoapUrl())) {
155                 result.addFieldError("jiraSoapUrl", cas.getText("configure.error.jirasoapurlempty"));
156                 result.setValid(false);
157             }
158 
159             if (ConfigUtil.isNullOrEmpty(config.getJiraSoapUsername())) {
160                 result.addFieldError("jiraSoapUsername", cas.getText("configure.error.jirasoapusernameempty"));
161                 result.setValid(false);
162             }
163 
164             String jiraSoapPassword = null;
165             if (config.getJiraSoapPassword()!=null) {
166                 jiraSoapPassword = config.getJiraSoapPassword();
167             }
168             else if (existingConfig != null && existingConfig.getJiraSoapPassword()!=null) {
169                 jiraSoapPassword = existingConfig.getJiraSoapPassword();
170             }
171             else {
172                 result.addFieldError("jiraSoapPassword", cas.getText("configure.error.jirasoappasswordnull"));
173                 result.setValid(false);
174             }
175 
176             // test connection
177             if (shouldTestExternalServices && result.isValid()) {
178                 try {
179                     JiraSoapServiceServiceLocator jiraSoapServiceGetter = new JiraSoapServiceServiceLocator();
180                     jiraSoapServiceGetter.setJirasoapserviceV2EndpointAddress(config.getJiraSoapUrl());
181                     JiraSoapService jiraSoapService = jiraSoapServiceGetter.getJirasoapserviceV2();
182 
183                     // Note: as long as we are ONLY logging in and logging out, don't need to save token and logout in finally
184                     String token = jiraSoapService.login(config.getJiraSoapUsername(), jiraSoapPassword);
185                     jiraSoapService.logout(token);
186                 }
187                 catch (Throwable t) {
188                     LogUtil.errorWithRemoteUserInfo(log, "Problem testing JIRA SOAP configuration by connecting to JIRA", t);
189                     result.addFieldError("jiraSoapUrl", cas.getText("configure.error.jirasoaptestconnectfailed") + ": " + t);
190                     result.setValid(false);
191                 }
192             }
193         }
194 
195         if (!ConfigUtil.isNotNullAndIsYesOrNo(config.getLdapAuthUsed())) {
196             result.addFieldError("ldapAuthUsed", cas.getText("configure.error.ldapauthusedinvalid"));
197             result.setValid(false);
198         }
199         else {
200             if ("YES".equals(config.getLdapAuthUsed())) {
201                 String providerType = config.getProviderType();
202                 if( providerType == null ||
203                         (!providerType.equals(CustomPermissionConfigConstants.PROVIDER_TYPE_OSUSER) &&
204                        !providerType.equals(CustomPermissionConfigConstants.PROVIDER_TYPE_ATLASSIAN_USER)))
205                 {
206                     result.addFieldError("providerType", cas.getText("configure.error.providertypeinvalid"));
207                     result.setValid(false);
208                 }
209                 else if (providerType.equals(CustomPermissionConfigConstants.PROVIDER_TYPE_OSUSER)) {
210                     if (config.getLdapUserIdAttribute()==null) {
211                         result.addFieldError("ldapUserIdAttribute", cas.getText("configure.error.ldapuseridattributenull"));
212                         result.setValid(false);
213                     }
214 
215                     if (config.getLdapEmailAttribute()==null) {
216                         result.addFieldError("ldapEmailAttribute", cas.getText("configure.error.ldapemailattributenull"));
217                         result.setValid(false);
218                     }
219 
220                     if (config.getLdapFirstNameAttribute()==null) {
221                         result.addFieldError("ldapFirstNameAttribute", cas.getText("configure.error.ldapfirstnameattributenull"));
222                         result.setValid(false);
223                     }
224 
225                     if (config.getLdapLastNameAttribute()==null) {
226                         result.addFieldError("ldapLastNameAttribute", cas.getText("configure.error.ldaplastnameattributenull"));
227                         result.setValid(false);
228                     }
229 
230                     if (config.getLdapProviderFullyQualifiedClassname()==null) {
231                         // ok to be empty. is not used in user-atlassian provider implementation
232                         result.addFieldError("ldapProviderFullyQualifiedClassname", cas.getText("configure.error.ldapproviderfullyqualifiedclassnamenull"));
233                         result.setValid(false);
234                     }
235                 }
236 
237                 String userFullNameFormat = config.getUserFullNameFormat();
238                 boolean lastCommaFirstFormat = (userFullNameFormat != null) && (userFullNameFormat.equals(CustomPermissionConfigConstants.USER_FULL_NAME_FORMAT_TYPE_LASTNAME_COMMA_FIRSTNAME));
239                 boolean firstLastFormat = (userFullNameFormat != null) && (userFullNameFormat.equals(CustomPermissionConfigConstants.USER_FULL_NAME_FORMAT_TYPE_FIRSTNAME_LASTNAME));
240                 boolean idFormat = (userFullNameFormat != null) && (userFullNameFormat.equals(CustomPermissionConfigConstants.USER_FULL_NAME_FORMAT_TYPE_ID));
241 
242                 if (userFullNameFormat==null) {
243                     result.addFieldError("userFullNameFormat", cas.getText("configure.error.userfullnameformatnull"));
244                     result.setValid(false);
245                 }
246                 else {
247                     if (!lastCommaFirstFormat && !firstLastFormat && !idFormat) {
248                         result.addFieldError("userFullNameFormat", cas.getText("configure.error.userfullnameformatinvalid") + ": " + userFullNameFormat);
249                         result.setValid(false);
250                     }
251                 }
252 
253                 String ldapConfigTestUsername = config.getLdapConfigTestUsername();
254                 if (StringUtil.isNullOrEmpty(ldapConfigTestUsername)) {
255                     result.addFieldError("ldapConfigTestUsername", cas.getText("configure.error.ldapconfigtestusernameempty"));
256                     result.setValid(false);
257                 }
258 
259                 // narrowing filter expression is configurable but is optional and don't even care if it is null or empty
260 
261                 if (shouldTestExternalServices && result.isValid())
262                 {
263                     try {
264                         LDAPUser usr = LDAPHelper.getLDAPUser(config, ldapConfigTestUsername);
265                         if(usr == null)
266                         {
267                             log.debug("Got null user back from LDAP for " + ldapConfigTestUsername);
268                             result.addFieldError("ldapAuthUsed", cas.getText("configure.error.ldapconfigtestreturnednull") + ": " + ldapConfigTestUsername);
269                             result.setValid(false);
270                         }
271                     }
272                     catch (Throwable t) {
273                         LogUtil.errorWithRemoteUserInfo(log, "Problem testing LDAP config in config UI", t);
274                         result.addFieldError("ldapAuthUsed", cas.getText("configure.error.ldapconfigtestfailure") + ": " + t.getMessage());
275                         result.setValid(false);
276                     }
277                 }
278             }
279         }
280 
281         if (!ConfigUtil.isNotNullAndIsYesOrNo(config.getUserSearchEnabled())) {
282             result.addFieldError("userSearchEnabled", cas.getText("configure.error.usersearchenabledinvalid"));
283             result.setValid(false);
284         }
285 
286         if (config.getMaxUserIDsLimit() == null || !ConfigUtil.isIntGreaterThanZero(config.getMaxUserIDsLimit())) {
287             result.addFieldError("maxUserIDsLimit", cas.getText("configure.error.maxuseridslimitinvalid"));
288             result.setValid(false);
289         }
290 
291         if (config.getMaxGroupIDsLimit() == null || !ConfigUtil.isIntGreaterThanZero(config.getMaxGroupIDsLimit())) {
292             result.addFieldError("maxGroupIDsLimit", cas.getText("configure.error.maxgroupidslimitinvalid"));
293             result.setValid(false);
294         }
295 
296         String pluginInDown = config.getPluginDown();
297         if (ConfigUtil.isNotNullAndIsYesOrNo(pluginInDown)) {
298             if ("YES".equals(pluginInDown)) {
299                 // is ok to be empty
300                 if (config.getDownTimeMessage() == null) {
301                     result.addFieldError("pluginInDown", cas.getText("configure.error.downtimemessagenull"));
302                     result.setValid(false);
303                 }
304             }
305         }
306         else {
307             result.addFieldError("pluginInDown", cas.getText("configure.error.plugindowninvalid"));
308             result.setValid(false);
309         }
310 
311         String groupActionsPermitted = config.getGroupActionsPermitted();
312         if (ConfigUtil.isNotNullAndIsYesOrNo(groupActionsPermitted)) {
313             if ("YES".equals(groupActionsPermitted)) {
314                 // these are ok to be empty
315                 if (config.getNewGroupNameCreationPrefixPattern() == null) {
316                     result.addFieldError("groupActionsPermitted", cas.getText("configure.error.newgroupnamecreationprefixpatterninvalid"));
317                     result.setValid(false);
318                 }
319 
320                 if (config.getNewGroupNameCreationSuffixPattern() == null) {
321                     result.addFieldError("groupActionsPermitted", cas.getText("configure.error.newgroupnamecreationsuffixpatterninvalid"));
322                     result.setValid(false);
323                 }
324             }
325         }
326         else {
327             result.addFieldError("groupActionsPermitted", cas.getText("configure.error.groupactionspermittedinvalid"));
328             result.setValid(false);
329         }
330 
331         if (!ConfigUtil.isNotNullAndIsYesOrNo(config.getPersonalSpaceAllowed())) {
332             result.addFieldError("personalSpaceAllowed", cas.getText("configure.error.personalspaceallowedinvalid"));
333             result.setValid(false);
334         }
335 
336         log.debug("CustomPermissionConfigAction - isValid=" + result + " fieldErrors=" + result.getFieldNameToErrorMessage());
337         return result;
338     }
