49  /***
50   * Action to configure plugin
51   * 
52   * @author Rajendra Kadam
53   * @author Gary S. Weaver
54   */
55  public class CustomPermissionConfigAction extends BaseCustomPermissionConfigAction implements Administrative
56  {
57      private Log log = LogFactory.getLog(this.getClass());
58  
59      BandanaManager bandanaManager;
60      BootstrapManager bootstrapManager;
61      CustomPermissionConfiguration customPermissionConfiguration;
62      private static final String JIRA_SOAP_PASSWORD_SET_PARAMNAME = "jiraSoapPasswordSet";
63      
64      
65      public CustomPermissionConfigAction()
66      {
67      	log.debug("CustomPermissionConfigAction instance created");
68      }
69      
70      public void setBandanaManager(BandanaManager bandanaManager)
71      {
72          this.bandanaManager = bandanaManager;
73      }
74  
75      public String doDefault() throws Exception
76      {
77          log.debug("CustomPermissionConfigAction - Inside doDefault ..");
78  
79  		LegacyConfigurationMigrator.migrateLegacyConfiguration(bandanaManager);
80  
81          configureFormValuesWithPersistedConfig();
82  
83          fixFormValues();
84  
85          // NOTE: the main reason for doing this and showing errors even though it might be your first time to the page
86          // is that people that have already configured the plugin once before will expect errors to show up on the page
87          // when they visit it, even if they didn't post to it yet.
88          if(!validateConfiguration()) {
89              return ERROR;
90          }
91  
92          return super.doDefault();
93      }
94  
95      public String execute() throws Exception
96      {
97  		log.debug("CustomPermissionConfigAction - Inside execute ..");
98  
99          // this is for logging logged in user info when logging error, fatal, warn
100         CacheUtil.setRemoteUser(getRemoteUser());
101 
102         List resultList = new ArrayList();
103 
104         fixFormValues();
105                 
106         if(!validateConfiguration()) {
107             LogUtil.warnWithRemoteUserInfo(log, "Configuration was invalid");
108             resultList.add(getText("configure.error.configurationinvalid"));
109             setActionErrors(resultList);
110 
111             return ERROR;
112         }
113 
114         updatePersistedConfigWithFormValues();
115 
116         resultList.add(getText("configure.success"));
117         setActionMessages(resultList);
118 
119         return SUCCESS;
120     }
121 
122     private void configureFormValuesWithPersistedConfig()
123     {
124         getCustomPermissionConfiguration().copyTo(this);
125     }
126 
127     private void updatePersistedConfigWithFormValues()
128     {        
129         getCustomPermissionConfiguration().updateWith(this);
130     }
131 
132     public void fixFormValues()
133     {
134         // Set presets as config, trimming and using defaults if they make sense
135         setDownTimeMessage(ConfigUtil.getTrimmedStringOrNull(getDownTimeMessage()));
136         setGroupActionsPermitted(ConfigUtil.getTrimmedStringOrUseDefaultIfValueIsNullOrTrimmedValueIsEmpty("groupActionsPermitted", getGroupActionsPermitted(), CustomPermissionConfigConstants.YES));
137         setMaxGroupIDsLimit("" + ConfigUtil.getIntOrUseDefaultIfNullOrTrimmedValueIsEmptyOrNotAnInteger("maxGroupIdsLimit", getMaxGroupIDsLimit(), 20));
138         setNewGroupNameCreationPrefixPattern(ConfigUtil.getTrimmedStringOrUseDefaultIfValueIsNullOrTrimmedValueIsEmpty("newGroupNameCreationPrefixPattern", getNewGroupNameCreationPrefixPattern(), CustomPermissionConstants.DEFAULT_NEW_GROUP_NAME_PREFIX));
139         setNewGroupNameCreationSuffixPattern(ConfigUtil.getTrimmedStringOrUseDefaultIfValueIsNullOrTrimmedValueIsEmpty("newGroupNameCreationSuffixPattern", getNewGroupNameCreationSuffixPattern(), CustomPermissionConstants.DEFAULT_NEW_GROUP_NAME_SUFFIX));
140         setMaxUserIDsLimit("" + ConfigUtil.getIntOrUseDefaultIfNullOrTrimmedValueIsEmptyOrNotAnInteger("maxUserIdsLimit", getMaxUserIDsLimit(), 20));
141         setPersonalSpaceAllowed(ConfigUtil.getTrimmedStringOrUseDefaultIfValueIsNullOrTrimmedValueIsEmpty("personalSpaceAllowed", getPersonalSpaceAllowed(), CustomPermissionConfigConstants.NO));
142         setPluginDown(ConfigUtil.getTrimmedStringOrUseDefaultIfValueIsNullOrTrimmedValueIsEmpty("pluginDown", getPluginDown(), CustomPermissionConfigConstants.NO));
143         setUserSearchEnabled(ConfigUtil.getTrimmedStringOrUseDefaultIfValueIsNullOrTrimmedValueIsEmpty("userSearchEnabled", getUserSearchEnabled(), CustomPermissionConfigConstants.YES));
144 
145         // only relevant for page itself, so not putting into context
146         Map paramMap = ServletActionContext.getRequest().getParameterMap();
147         log.debug("paramMap: " + paramMap);
148         if (paramMap.get(JIRA_SOAP_PASSWORD_SET_PARAMNAME)==null) {
149             // make sure password is set to null if jiraSoapPasswordSet is not checked/set
150             setJiraSoapPassword(null);
151         }
152     }
153     
154     public boolean validateConfiguration()
155 	{
156 		log.debug("CustomPermissionConfigAction - Inside validate Configuration ...");
157         CsumConfigValidationResponse validResp = CustomPermissionConfiguration.validate(this, getCustomPermissionConfiguration(), getRemoteUser().getName(), this, true);
158         Map fieldErrorMap = validResp.getFieldNameToErrorMessage();
159         Iterator keys = fieldErrorMap.keySet().iterator();
160         while (keys.hasNext()) {
161             String key = (String)keys.next();
162             addFieldError(key, (String)fieldErrorMap.get(key));
163         }
164 
165         return validResp.isValid();
166     }
